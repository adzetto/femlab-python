\documentclass[tikz, border=15pt]{standalone}
\usepackage{amsmath}
\usetikzlibrary{calc, angles, quotes, positioning, arrows.meta}

\begin{document}

\begin{tikzpicture}[scale=1.4, >=stealth, font=\sffamily]

    % Define Coordinates
    \coordinate (N1) at (0, 0);
    \coordinate (N2) at (6, 0);
    \coordinate (N3) at (3, 2.64575);

    % Styles
    \tikzset{
        feLine/.style={thick},
        feNode/.style={circle, draw, fill=white, inner sep=1.5pt, minimum size=4pt},
        nodeLabel/.style={circle, draw=black, fill=white, inner sep=1pt, minimum size=15pt, font=\small},
        dof/.style={->, thin},
        support line/.style={thick}
    }

    % Custom Macro for Element Labels
    % #1 = Start Node, #2 = End Node, #3 = Element Number
    \newcommand{\drawElement}[3]{
        \draw[feLine] (#1) -- (#2);
        \path (#1) -- (#2) node[midway, sloped, anchor=south, draw, fill=white, inner sep=2pt, minimum size=14pt, font=\footnotesize] (E#3) {#3};
        \coordinate (TopLeft#3) at ($(E#3.south west)!1.25!(E#3.north west)$);
        \coordinate (TopRight#3) at ($(E#3.south east)!1.25!(E#3.north east)$);
        \draw[-{Stealth[length=2.5mm, width=1.5mm]}, semithick] (TopLeft#3) -- (TopRight#3);
    }

    % Global Coordinate System
    \begin{scope}[shift={(-1.2, -0.5)}]
        \draw[->] (0,0) -- (0.6, 0) node[right, font=\scriptsize] {$x$};
        \draw[->] (0,0) -- (0, 0.6) node[above, font=\scriptsize] {$y$};
    \end{scope}

    % Node 1: Pinned Support
    \begin{scope}[shift={(N1)}]
        \draw[support line] (0, -0.1) -- (-0.3, -0.5) -- (0.3, -0.5) -- cycle;
        \draw[support line] (-0.4, -0.5) -- (0.4, -0.5);
        \foreach \x in {-0.3, -0.1, 0.1, 0.3} {
            \draw (\x, -0.5) -- ++(-0.15, -0.2);
        }
    \end{scope}

    % Node 2: Inclined Roller Support
    \begin{scope}[shift={(N2)}, rotate=60]
        \draw[support line] (0, -0.1) -- (-0.3, -0.5) -- (0.3, -0.5) -- cycle;
        \draw[thick] (-0.15, -0.58) circle (0.08);
        \draw[thick] (0.15, -0.58) circle (0.08);
        \draw[support line] (-0.5, -0.66) -- (0.5, -0.66);
        \foreach \x in {-0.4, -0.2, 0, 0.2, 0.4} {
            \draw (\x, -0.66) -- ++(-0.15, -0.2);
        }
    \end{scope}

    % Angle dimension for the 60-degree roller ground
    \coordinate (G_origin) at ($(N2) + (-30:0.96)$);
    \draw[dashed] (G_origin) -- ++(1, 0) coordinate (G_horiz);
    \draw[dashed] (G_origin) -- ++(60:1) coordinate (G_plane);
    \pic [draw, "$\,60^\circ$", angle radius=0.5cm, angle eccentricity=1.5, font=\scriptsize] {angle = G_horiz--G_origin--G_plane};

    % Draw Elements using the custom macro
    \drawElement{N1}{N3}{1}
    \drawElement{N3}{N2}{2}
    \drawElement{N1}{N2}{3}

    % Draw Nodes (circles at joints)
    \node[feNode] at (N1) {};
    \node[feNode] at (N2) {};
    \node[feNode] at (N3) {};

    % Node Labels (circled numbers)
    \node[nodeLabel] at ($(N1)+(-0.35, 0.25)$) {1};
    \node[nodeLabel] at ($(N2)+(0.35, 0.25)$) {2};
    \node[nodeLabel] at ($(N3)+(-0.4, 0.2)$) {3};

    % Material and Geometry Annotations
    \node[anchor=north west, align=left, font=\footnotesize] at (-0.5, 3.2) {
        $A = 1 \text{ mm}^2$ \\
        $E = 64 \text{ MPa}$
    };
    \node[font=\footnotesize, rotate=41.4] at ($(N1)!0.5!(N3) + (-0.35, 0.6)$) {$L=4$};
    \node[font=\footnotesize, rotate=-41.4] at ($(N2)!0.5!(N3) + (0.35, 0.6)$) {$L=4$};

    % L = 6 Dimension Line
    \draw (N1) ++(0, -0.9) -- ++(0, -0.4);
    \draw (N2) ++(0, -0.9) -- ++(0, -0.4);
    \draw[<->] ($(N1) + (0, -1.1)$) -- ($(N2) + (0, -1.1)$) node[midway, fill=white, font=\footnotesize] {$L=6$};

    % External Force
    \draw[->, thick] (N3) ++(0, 1.2) -- ++(0, -1.0) node[pos=0, above] {$P = 10\text{ N}$};

    % Degrees of Freedom (DOFs)
    \begin{scope}[every node/.style={font=\footnotesize}]
        \draw[dof] (N1) ++(0.2, 0) -- ++(0.6, 0) node[below] {$u_1$};
        \draw[dof] (N1) ++(0, 0.2) -- ++(0, 0.6) node[above] {$u_2$};

        \draw[dof] (N2) ++(0.2, 0) -- ++(0.6, 0) node[right] {$u_3$};
        \draw[dof] (N2) ++(0, 0.2) -- ++(0, 0.6) node[above] {$u_4$};

        \draw[dof] (N3) ++(0.2, 0) -- ++(0.6, 0) node[right] {$u_5$};
        \draw[dof] (N3) ++(0.15, 0.15) -- ++(0, 0.6) node[above] {$u_6$};
    \end{scope}

\end{tikzpicture}

\end{document}
